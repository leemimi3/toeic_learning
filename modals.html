📦 方案一：手動備份（JSON）
把所有單字和筆記下載成一個 .json 檔案存到電腦，需要時再匯入還原。
📊 方案二：Google Sheets 同步
資料自動備份到你的 Google 試算表。換裝置開網站時會自動偵測並還原，不需要手動操作。
⚙ 設定步驟（只需做一次）
  1. 開啟 script.google.com，點「新增專案」
  2. 把下方程式碼貼入編輯器並儲存
  3. 點「部署 → 新增部署」，類型選「網路應用程式」
  4. 執行身分選「我」，存取權選「任何人」，按「部署」
  5. 複製產生的 Web App URL，貼到下方欄位
展開 Apps Script 程式碼 
const SHEET_NAME = 'toeic-backup';

function doPost(e) {
  const data = JSON.parse(e.postData.contents);
  const ss   = SpreadsheetApp.getActiveSpreadsheet();
  let sheet  = ss.getSheetByName(SHEET_NAME);
  if (!sheet) sheet = ss.insertSheet(SHEET_NAME);
  sheet.clearContents();
  sheet.getRange(1,1).setValue(data.vocab  || '[]');
  sheet.getRange(2,1).setValue(data.notes  || '[]');
  sheet.getRange(3,1).setValue(data.streak || '[]');
  sheet.getRange(4,1).setValue(data.syncAt || '');
  return ContentService.createTextOutput('ok');
}

function doGet(e) {
  const ss    = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) return ContentService.createTextOutput(JSON.stringify({})).setMimeType(ContentService.MimeType.JSON);
  const result = {
    vocab:  sheet.getRange(1,1).getValue(),
    notes:  sheet.getRange(2,1).getValue(),
    streak: sheet.getRange(3,1).getValue(),
    syncAt: sheet.getRange(4,1).getValue(),
  };
  return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON);
}
